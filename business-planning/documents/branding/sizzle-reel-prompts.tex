\section{Music and Sound Direction}

\begin{tcolorbox}[dreambox, title=Audio Guidelines for AI Music Generation]
\textbf{Overall Feel}: Modern, optimistic, innovative without being cold or corporate

\textbf{Structure}:
- Opening: Ambient, building anticipation (0-10s)
- First crescendo: Energy builds with AI showcase (10-30s)
- Dynamic middle: Rhythmic, showcasing integration (30-90s)
- Emotional peak: Client success stories (90-110s)
- Final build: Inspiring, future-focused (110-120s)

\textbf{Elements to Include}:
- Electronic base with organic instruments
- Subtle use of vocoder/synthesis for tech feel
- Building percussion representing progress
- Melodic hooks that feel innovative yet accessible
- Sound design elements: data flows, connection sounds, transformation wooshes

\textbf{Reference Tracks}:
- M83 - ``Outro'' (epic, building)
- Daft Punk - ``End of Line'' (tech feel)
- Hans Zimmer - ``Time'' (emotional depth)
- The Social Network Soundtrack (innovation vibe)
\end{tcolorbox}

\section{Technical Specifications}

\begin{tcolorbox}[dreambox, title=Output Requirements]
\textbf{Resolution}: 4K (3840 x 2160) minimum, 8K preferred for future-proofing

\textbf{Frame Rate}: 60fps for smooth motion, 24fps for cinematic sequences

\textbf{Colour Space}: Rec. 709 for broadcast, DCI-P3 for digital cinema

\textbf{Format}: ProRes 4444 XQ or DNxHR 444 for maximum quality

\textbf{Duration}: 
- Full sizzle reel: 2 minutes 30 seconds
- Short version: 60 seconds
- Social cuts: 15 and 30 seconds
- Loop version: 3 minutes for events

\textbf{Aspect Ratios}:
- 16:9 for primary use
- 1:1 for social media
- 9:16 for stories/reels
- 21:9 for ultra-wide displays
\end{tcolorbox}

\section{Additional Notes for Production}

\begin{tcolorbox}[dreambox, title=Production Considerations]
\textbf{Brand Consistency}:
- Ensure colour accuracy for all brand colours
- Maintain consistent lighting temperature
- Use motion blur carefully to maintain clarity
- Keep text legible at all sizes

\textbf{Accessibility}:
- Ensure sufficient contrast in all scenes
- Avoid rapid flashing or strobing effects
- Include captions for any spoken content
- Design for colour-blind viewers

\textbf{Legal Considerations}:
- Ensure all generated content is clearly artificial
- No recognizable real people without consent
- Avoid branded products or logos
- Keep architectural representations respectful

\textbf{Future Flexibility}:
- Create modular sequences for easy updates
- Maintain project files for re-rendering
- Design with multiple end-uses in mind
- Build in space for partner/client logos
\end{tcolorbox}
