\section{Crisis Communications}

\subsection{Crisis Preparedness}

\begin{tcolorbox}[prbox, title=Crisis Response Framework]
\textbf{Potential Scenarios}:
\begin{itemize}
    \item Data breach or security incident
    \item Project failure or client complaint
    \item Staff misconduct or departure
    \item Technology malfunction
    \item Negative social media campaign
\end{itemize}

\textbf{Response Protocol}:
\begin{enumerate}
    \item Assess situation (30 mins)
    \item Convene crisis team (1 hour)
    \item Draft holding statement (2 hours)
    \item Full response within 24 hours
    \item Follow-up communications plan
\end{enumerate}
\end{tcolorbox}

\section{Crisis Communications Templates}

\subsection{Holding Statements}

\begin{tcolorbox}[prbox, title=Template Holding Statements]
\textbf{General}:
``We are aware of [situation] and are investigating as a matter of priority. We will provide a full update as soon as we have all the facts.''

\textbf{Client Issue}:
``Client satisfaction is our top priority. We are working directly with [client] to resolve this matter swiftly and to their complete satisfaction.''

\textbf{Technical Issue}:
``We have identified [issue] affecting [scope]. Our technical team is working around the clock to resolve this, and we expect normal service to resume by [time].''
\end{tcolorbox}
