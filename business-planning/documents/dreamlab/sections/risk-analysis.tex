\subsection{Critical Risk Assessment}

\subsubsection{Market and Commercial Risks}

\begin{riskbox}
\textbf{Risk}: Slower than expected AI training market adoption\\
\textbf{Impact}: High - Could reduce revenue by 30-40\%\\
\textbf{Probability}: Medium\\
\textbf{Mitigation Strategies}:
\begin{itemize}
    \item Diversify into data science and analytics training
    \item Develop entry-level programmes for SMEs
    \item Create government-funded apprenticeship schemes
    \item Expand online offering to reach wider market
\end{itemize}
\end{riskbox}

\subsubsection{Technology Risks}

\begin{riskbox}
\textbf{Risk}: Rapid hardware obsolescence requiring early refresh\\
\textbf{Impact}: High - £200k unplanned capital expenditure\\
\textbf{Probability}: Medium\\
\textbf{Mitigation Strategies}:
\begin{itemize}
    \item Lease vs buy strategy for some equipment
    \item Vendor partnership for upgrade programmes
    \item Cloud hybrid approach for peak capacity
    \item Equipment refresh reserve fund (5\% revenue)
\end{itemize}
\end{riskbox}

\subsection{Financial Risk Management}

\subsubsection{Cash Flow Risk}

\begin{itemize}
    \item \textbf{Risk}: Corporate clients delaying payment (60-90 days)
    \item \textbf{Impact}: Working capital strain
    \item \textbf{Mitigation}: 
    \begin{itemize}
        \item Invoice factoring facility (£100k)
        \item 50\% deposit requirement
        \item Credit insurance for large accounts
        \item Diversified client base (no client >15\% revenue)
    \end{itemize}
\end{itemize}

\subsubsection{Currency Risk}

\begin{itemize}
    \item \textbf{Risk}: USD equipment costs vs GBP revenue
    \item \textbf{Impact}: 10-15\% cost variance potential
    \item \textbf{Mitigation}: Forward contracts for major purchases
\end{itemize}
